\section*{Всякие разные неравенства}

% $authors:
% - Владимир Викторович Трушков

\begin{problems}

\item
Для положительных чисел $a_{1}$, $a_{2}$, \ldots, $a_{n}$
и~$b_{1}$, $b_{2}$, \ldots, $b_{n}$ докажите неравенство
\begin{problemeq}
    \frac{a_{1}^2}{b_{1}} + \frac{a_{2}^2}{b_{2}}
    + \ldots +
    \frac{a_{n}^2}{b_{n}}
\geq
    \frac{
        (a_{1} + a_{2} + \ldots + a_{n})^2
    }{
        b_{1} + b_{2} + \ldots + b_{n}
    }
\end{problemeq}.

\item
Пусть $a, b, c > 0$.
Докажите неравенство
\begin{problemeq}
    \frac{a}{b + 2 c} + \frac{b}{c + 2 a} + \frac{c}{a + 2 b}
\geq
    1
\end{problemeq}.

\item
\emph{Неравенство Чебышёва.}
Докажите, что для неотрицательных чисел
$a_{1} \geq a_{2} \geq \ldots \geq a_{n}$,
$b_{1} \geq b_{2} \geq \ldots \geq b_{n}$ справедливо неравенство
\begin{problemeq}
    (a_{1} + a_{2} + \ldots + a_{n})
    \cdot
    (b_{1} + b_{2} + \ldots + b_{n})
\leq
    n \cdot (a_{1} b_{1} + a_{2} b_{2} + \ldots + a_{n} b_{n})
\end{problemeq}.

\item
Докажите, что для положительных чисел $a$, $b$, $c$, $d$ выполнено неравенство
\begin{problemeq}
    \frac{a + b + c + d}{a b c d}
\leq
    \frac{1}{a^3} + \frac{1}{b^3} + \frac{1}{c^3} + \frac{1}{d^3}
\end{problemeq}.

\item
Докажите для положительных $a$, $b$, $c$ неравенство
\begin{problemeq}
    a b c
\geq
    (b + c - a) \cdot (c + a - b) \cdot (a + b - c)
\end{problemeq}.

\item
Сумма положительных $a$, $b$, $c$, $d$ равна $4$.
Докажите, что
\begin{problemeq}
    \frac{a}{b^2 + 1} + \frac{b}{c^2 + 1} +
    \frac{c}{d^2 + 1} + \frac{d}{a^2 + 1}
\geq
    2
\end{problemeq}.

\item
Известно, что $a, b, c > 0$, $a b + b c + c a = 3$.
Докажите, что
\begin{problemeq}
    \frac{a b}{1 + c^2} + \frac{b c}{1 + a^2} + \frac{c a}{1 + b^2}
\geq
    \frac{3}{2}
\end{problemeq}.

\item
Для положительных чисел $a_{1}$, $a_{2}$, \ldots, $a_{n}$ таких, что
$a_{1} + a_{2} + \ldots + a_{n} = 1$, докажите неравенство
\begin{problemeq}
    \left( a_{1} + \frac{1}{a_{1}} \right)^2 +
    \left( a_{2} + \frac{1}{a_{2}} \right)^2
    + \ldots +
    \left( a_{n} + \frac{1}{a_{n}} \right)^2
\geq
    \frac{(n^2 + 1)^2}{n}
\end{problemeq}.

\item
Пусть $a, b, c > 0$.
Докажите неравенство
\begin{problemeq}
    \frac{a^3}{a^2 + a b + b^2} + \frac{b^3}{b^2 + b c + c^2} +
    \frac{c^3}{c^2 + c a + c^2}
\geq
    \frac{a + b + c}{3}
\end{problemeq}.

\item
Сумма положительных чисел $a$, $b$, $c$, $d$ равна $4$.
Докажите, что
\begin{problemeq}
    \frac{a}{a^3 + 4} + \frac{b}{b^3 + 4} +
    \frac{c}{c^3 + 4} + \frac{d}{d^3 + 4}
\leq
    \frac{4}{5}
\end{problemeq}.

\end{problems}
